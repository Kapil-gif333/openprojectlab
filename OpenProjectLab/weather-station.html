Weather Station

Components

  • DHT11 sensor
  • ESP8266

Code

#include 

#define DHTPIN D4
#define DHTTYPE DHT11

DHT dht(DHTPIN, DHTTYPE);

void setup(){
Serial.begin(9600);
dht.begin();
}

void loop(){
float t = dht.readTemperature();
float h = dht.readHumidity();

Serial.print("Temp:");
Serial.println(t);

Serial.print("Humidity:");
Serial.println(h);

delay(2000);
}

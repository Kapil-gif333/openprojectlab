Weather Station Project

Monitor temperature and humidity using a DHT11 sensor and Arduino.

Components Required

  • Arduino Uno
  • DHT11 Sensor
  • 16x2 LCD Display (Optional)
  • Breadboard
  • Jumper Wires

Connections

Component Arduino Pin
DHT11 Data D2
VCC 5V
GND GND

Arduino Code

#include 

#define DHTPIN 2
#define DHTTYPE DHT11

DHT dht(DHTPIN, DHTTYPE);

void setup() {

Serial.begin(9600);
dht.begin();

}

void loop() {

float temp = dht.readTemperature();
float hum = dht.readHumidity();

Serial.print("Temperature: ");
Serial.print(temp);
Serial.println(" C");

Serial.print("Humidity: ");
Serial.print(hum);
Serial.println(" %");

delay(2000);

}

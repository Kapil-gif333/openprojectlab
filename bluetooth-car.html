💧 Smart Plant Irrigation System

An automatic agricultural loop monitoring soil hydration levels via an analog soil probe and triggering a submersible water pump when fields dry up.

🛠️ Step-by-Step Construction Guide

  1. Push the two-pronged soil moisture probe firmly into your target plant monitoring earth block base.
  2. Connect the probe controller board analog output pin over to analog input pin A0 on your Arduino.
  3. Cut the positive wire of your water pump and route it through the Normally Open (NO) and Common (COM) pins of your relay.
  4. Connect the relay signal trigger wire directly to digital pin 7 on your board.
  5. Set up a test threshold index inside your code script, place the sensor in dry soil, and check if the water pump kicks on automatically.

const int soilSensorPin = A0;
const int relayControlPin = 7;
int moistureValue = 0;
int dryThreshold = 700; // Calibrate this value based on your soil type

void setup() {
  pinMode(relayControlPin, OUTPUT);
  digitalWrite(relayControlPin, HIGH); // Keeps pump closed by default
  Serial.begin(9600);
}

void loop() {
  moistureValue = analogRead(soilSensorPin);
  Serial.print("Soil Moisture Value Tracker: ");
  Serial.println(moistureValue);
  
  if (moistureValue > dryThreshold) { 
    // Higher resistance values mean dry conditions are met
    digitalWrite(relayControlPin, LOW);  // Turn on water pump module
  } else {
    digitalWrite(relayControlPin, HIGH); // Shut down water pump module
  }
  delay(1000); // Check soil moisture balance levels every second
}
